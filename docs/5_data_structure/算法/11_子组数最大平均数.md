# 11、子数组的最大平均数

```js
  // 求子数组的平均数最大
```

```js
// 方式一： 使用滑动窗口的方法
  function findMaxAverage(arr, k) {
    let sum = 0;
    for(let i = 0; i < k; i++) {
      sum += arr[i]
    }

    let max = sum;
    for(let i = k; i < arr.length; i++) {
      sum += arr[i] - arr[i - k];
      max = Math.max(max, sum)
    }

    return max/k
  }
  console.log('findMaxAverage--', findMaxAverage([2, 3, 4, 5, 6, 7, 8, 9], 2));
```
