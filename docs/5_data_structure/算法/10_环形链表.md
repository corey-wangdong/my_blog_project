// 方式一： 快慢指针方式
function hasCircle(head) {
  if(head == null || head.next == null) return false;

  let slow = head, fast = head.next

  while(fast !== null && fast.next !== null) {
    slow = slow.next;
    fast = fast.next.next;

    if(slow == fast)  return true;
  }

  return false
}
